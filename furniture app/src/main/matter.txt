# 7. Auth Service (Login / Signup Logic)

📦 `services/AuthService.java`

```java
package services;

import users.Customer;
import java.util.*;

public class AuthService {

    List<Customer> customers = new ArrayList<>();

    public Customer signup(String username,String password){

        Customer c = new Customer(username,password);
        customers.add(c);

        System.out.println("Signup Successful");

        return c;
    }

    public Customer login(String username,String password){

        for(Customer c:customers){

            if(c.getUsername().equals(username)){
                System.out.println("Login Successful");
                return c;
            }
        }

        System.out.println("User not found");
        return null;
    }

}
```

# 8. Product Service (Store Products)

📦 `services/ProductService.java`

```java
package services;

import models.*;
import java.util.*;

public class ProductService {

    List<Furniture> products = new ArrayList<>();

    public ProductService(){

        products.add(new Chair("Office Chair",3000));
        products.add(new Chair("Gaming Chair",7000));

        products.add(new Sofa("Luxury Sofa",15000));

        products.add(new Bed("Single Bed",9000));
        products.add(new Bed("Double Bed",16000));

        products.add(new Locker("Single Door Locker",8000));
        products.add(new Locker("Double Door Locker",12000));

        products.add(new Almar("Wooden Almar",11000));

    }

    public void showProducts(){

        int i=1;

        for(Furniture f:products){
            System.out.print(i+". ");
            f.display();
            i++;
        }

    }

    public Furniture getProduct(int index){

        if(index>0 && index<=products.size()){
            return products.get(index-1);
        }

        return null;
    }

}
```
